修改服务器端服务配置 ​

  • 修改frps.ini 配置文件
txt
[common]
bind_port = 7000 # 绑定服务器端口  如果端口没有开启，请先开启端口
vhost_http_port = 6125 # 暴露出去的端口  可以给nginx使用代理
