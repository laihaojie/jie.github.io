eslint 配置 ​

json
{
  "editor.formatOnSave": true,
  "editor.formatOnType": true,
  "eslint.codeAction.showDocumentation": {
    "enable": true
  },
  "editor.codeActionsOnSave": {
    "source.fixAll": true
  },
  "editor.tabSize": 2,
  "eslint.validate": [
    "javascript",
    "javascriptreact",
    "typescript",
    "javascriptreact",
    "html",
    "typescriptreact",
    "vue"
  ]
}

json
{
  "vim.handleKeys": {
    "<C-c>": false,
    "<C-a>": false,
    "<C-x>": false,
    "<C-f>": false,
    "<C-h>": false,
    "<C-s>": false,
    "<C-z>": false,
    "<C-y>": false

  "vim.normalModeKeyBindings": [
    {
      "before": ["H"],
      "after": ["^"]
    },
    {
      "before": ["L"],
      "after": ["g", "_"]
    }
  ],
  "vim.operatorPendingModeKeyBindings": [
    {
      "before": ["H"],
      "after": ["^"]
    },
    {
      "before": ["L"],
      "after": ["g", "_"]
    }
  ]
}
