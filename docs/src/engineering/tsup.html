tsup 配置 ​

ts
import type { Options } from 'tsup'

export const tsup: Options = {
  entry: [
    'src/*.ts',
  ],
  format: ['cjs', 'esm'],
  dts: true,
  splitting: true,
  clean: true,
  shims: true,
  // 跳过 node_modules 打包
  // skipNodeModulesBundle: true,
  // 跳过指定 package 打包
  external: ['miniprogram-ci']
}
